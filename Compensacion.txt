      function calculo() {
        var datos = parseInt(document.getElementById("datos").value);
        var minutos = parseInt(document.getElementById("minutos").value);
        var mensajes = parseInt(document.getElementById("mensajes").value);
        var dias = parseInt(document.getElementById("dias").value)*24;
        if(dias>0){
          var resultadoDatos =  parseInt(2*((datos/720)*dias));
          var resultadoMinutos =  parseInt(2*((minutos/720)*dias));
          var resultadoMensajes =  parseInt(2*((mensajes/720)*dias));
        } else {
          alert("Ingrese los dias a compensar");
        }
        document.getElementById("resultado").innerHTML = "Se debe compensar al usuario as&iacute;: Datos - "+resultadoDatos+"; Minutos - "+resultadoMinutos+"; Mensajes -"+resultadoMensajes;
      }
